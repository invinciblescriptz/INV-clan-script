-- AUTO RELOAD TOOL FUNCTION
local function startAutoRep(toolName, toggleVar)
    task.spawn(function()
        while _G[toggleVar] do
            pcall(function()
                local char = LocalPlayer.Character
                if not char or not char:FindFirstChild("Humanoid") or char.Humanoid.Health <= 0 then
                    task.wait(1.5)
                    return
                end
                local tool = LocalPlayer.Backpack:FindFirstChild(toolName) or char:FindFirstChild(toolName)
                if tool and tool.Parent ~= char then
                    char.Humanoid:EquipTool(tool)
                end
                if muscleEvent then
                    muscleEvent:FireServer("rep")
                end
            end)
            task.wait(0.025)
        end
        -- Unequip tool when toggled off
        pcall(function()
            local char = LocalPlayer.Character
            if char then
                local eq = char:FindFirstChild(toolName)
                if eq then eq.Parent = LocalPlayer.Backpack end
            end
        end)
    end)
end

-- KILL TAB
local KillTab = window:AddTab("Kill")
local KillFolder = KillTab:AddFolder("Auto Kill")

local function checkCharacter()
    if not LocalPlayer.Character then
        LocalPlayer.CharacterAdded:Wait()
    end
    return LocalPlayer.Character
end

local function killPlayer(target)
    local char = checkCharacter()
    if not char or not char:FindFirstChild("LeftHand") then return end
    local tChar = target.Character
    if tChar and tChar:FindFirstChild("HumanoidRootPart") and tChar:FindFirstChild("Humanoid") and tChar.Humanoid.Health > 0 then
        firetouchinterest(tChar.HumanoidRootPart, char.LeftHand, 0)
        firetouchinterest(tChar.HumanoidRootPart, char.LeftHand, 1)
        local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or char:FindFirstChild("Punch")
        if punch and char:FindFirstChild("Humanoid") then
            char.Humanoid:EquipTool(punch)
        end
        if muscleEvent then
            muscleEvent:FireServer("punch", "leftHand")
            muscleEvent:FireServer("punch", "rightHand")
        end
    end
end

local rockOptions = {
    {name = "Tiny Island Rock", dur = 0},
    {name = "Starter Island Rock", dur = 100},
    {name = "Legend Beach Rock", dur = 5000},
    {name = "Frost Gym Rock", dur = 150000},
    {name = "Mythical Gym Rock", dur = 400000},
    {name = "Eternal Gym Rock", dur = 750000},
    {name = "Legend Gym Rock", dur = 1000000},
    {name = "Muscle King Gym Rock", dur = 5000000},
    {name = "Ancient Jungle Rock", dur = 10000000},
}

local selectedDur = nil
local rockDropdown = RockFarmFolder:AddDropdown("Select Rock", {
    Title = "Select Rock to Farm",
    Values = (function()
        local vals = {"None"}
        for _, opt in ipairs(rockOptions) do
            table.insert(vals, opt.name)
        end
        return vals
    end)(),
    Callback = function(selected)
        selectedDur = nil
        for _, opt in ipairs(rockOptions) do
            if opt.name == selected then
                selectedDur = opt.dur
                break
            end
        end
        print("Selected rock durability: " .. (selectedDur or "None"))
    end,
})

local function equipPunch()
    pcall(function()
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("Humanoid") then
            for _, tool in pairs(LocalPlayer.Backpack:GetChildren()) do
                if tool.Name == "Punch" then
                    char.Humanoid:EquipTool(tool)
                end
            end
            if muscleEvent then
                muscleEvent:FireServer("punch", "leftHand")
                muscleEvent:FireServer("punch", "rightHand")
            end
        end
    end)
end

local autoRockFarm = false
local autoFarmConnection = nil

RockFarmFolder:AddSwitch("Auto Farm Selected Rock", function(state)
    getgenv().autoRockFarm = state
    if state and selectedDur then
        autoFarmConnection = task.spawn(function()
            while getgenv().autoRockFarm do
                pcall(function()
                    local durVal = LocalPlayer:FindFirstChild("Durability")
                    if durVal and durVal.Value >= selectedDur then
                        for _, desc in pairs(workspace:FindFirstChild("machinesFolder") and workspace.machinesFolder:GetDescendants() or {}) do
                            if desc.Name == "neededDurability" and desc.Value == selectedDur then
                                local rock = desc.Parent:FindFirstChild("Rock")
                                if rock then
                                    local char = LocalPlayer.Character
                                    if char and char:FindFirstChild("RightHand") and char:FindFirstChild("LeftHand") then
                                        firetouchinterest(rock, char.RightHand, 0)
                                        firetouchinterest(rock, char.RightHand, 1)
                                        firetouchinterest(rock, char.LeftHand, 0)
                                        firetouchinterest(rock, char.LeftHand, 1)
                                        equipPunch()
                                    end
                                end
                            end
                        end
                    end
                end)
                task.wait(0.15)
            end
        end)
        print("Rock farm started for durability >= " .. selectedDur)
    else
        print("Rock farm stopped or no rock selected")
    end
end)
