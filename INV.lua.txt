-- Helper Functions
local function getHumanoidRootPart()
    local character = LocalPlayer.Character
    return character and character:FindFirstChild(GAMECONSTANTS.HUMANOIDROOTPARTNAME)
end

local function getHumanoid()
    local character = LocalPlayer.Character
    return character and character:FindFirstChildOfClass(GAMECONSTANTS.HUMANOIDNAME)
end

local function tpTo(position)
    local hrp = getHumanoidRootPart()
    if hrp and position then
        hrp.CFrame = CFrame.new(position)
    end
end

local function safeEquip(toolName)
    pcall(function()
        local character = LocalPlayer.Character
        local humanoid = getHumanoid()
        if not character or not humanoid or humanoid.Health <= 0 then return end

        local tool = LocalPlayer.Backpack:FindFirstChild(toolName) or character:FindFirstChild(toolName)
        if tool and tool.Parent ~= character then
            character.Humanoid:EquipTool(tool)
        end
    end)
end

-- Auto-Rep (Pushups / Weights / Handstands)
local function startAutoRep(toolName, stateKey)
    task.spawn(function()
        while states[stateKey] and NexusRunning do
            pcall(function()
                local character = LocalPlayer.Character
                local humanoid = getHumanoid()
                if not humanoid or humanoid.Health <= 0 then task.wait(2); return end -- Wait if dead or no humanoid

                safeEquip(toolName)
                local muscleEvent = LocalPlayer:WaitForChild(GAMECONSTANTS.MUSCLEEVENTNAME, 5) -- Wait for event
                if muscleEvent then
                    muscleEvent:FireServer("rep")
                else
                    warn("[INV] muscleEvent not found during auto-rep.")
                    task.wait(1) -- Wait longer if event is missing
                end
            end)
            task.wait(0.1) -- Small delay between actions
        end
        -- Cleanup: Unequip tool if it was equipped by the script
        pcall(function()
            local character = LocalPlayer.Character
            if character then
                local tool = character:FindFirstChild(toolName)
                if tool then tool.Parent = LocalPlayer.Backpack end
            end
        end)
    end)
end

-- Infinite Jump
local jumpConnection = nil
local function toggleInfiniteJump(enabled)
    states.InfiniteJump = enabled
    if enabled then
        if jumpConnection then return end -- Already connected
        jumpConnection = UIS.JumpRequest:Connect(function()
            pcall(function()
                local humanoid = getHumanoid()
                if humanoid then
                    humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
                end
            end)
        end)
        addConnection(jumpConnection)
    else
        if jumpConnection then
            jumpConnection:Disconnect()
            jumpConnection = nil
        end
    end
end

end

task.spawn(function()
    while NexusRunning do
        if rockFarmActive and selectedRockDurability > 0 then
            pcall(function()
                local character = LocalPlayer.Character
                local humanoid = getHumanoid()
                if not humanoid or humanoid.Health <= 0 then task.wait(1.5); return end -- Wait if dead

                local durVal = LocalPlayer:FindFirstChild(GAMECONSTANTS.DURABILITYVALUENAME)
                if not durVal or durVal.Value < selectedRockDurability then
                    task.wait(0.8); return -- Wait if durability is too low
                end

                safeEquip(GAMECONSTANTS.PUNCHTOOLNAME)
                local machinesFolder = WS:FindFirstChild(GAMECONSTANTS.MACHINESFOLDERNAME)
                if machinesFolder then
                    for , obj in ipairs(machinesFolder:GetDescendants()) do
                        if obj.Name == GAMECONSTANTS.NEEDEDDURABILITYNAME and obj:IsA("IntValue") and obj.Value == selectedRockDurability then
                            local rock = obj.Parent:FindFirstChild(GAMECONSTANTS.ROCKNAME)
                            if rock and rock:IsA("BasePart") then
                                local hrp = getHumanoidRootPart()
                                if hrp then
                                    -- Simulate touching the rock to initiate interaction
                                    firetouchinterest(rock, hrp, 0) -- Start touch
                                    task.wait(0.025)
                                    firetouchinterest(rock, hrp, 1) -- End touch
                                    task.wait(0.05) -- Small delay before firing events

                                    local muscleEvent = LocalPlayer:WaitForChild(GAMECONSTANTS.MUSCLEEVENTNAME, 5)
                                    if muscleEvent then
                                        muscleEvent:FireServer("punch", "leftHand")
                                        muscleEvent:FireServer("punch", "rightHand")
                                    else
                                        warn("[INV] muscleEvent not found during rock farm.")
                                    end
                                end
                            end
                            break -- Found the correct rock, exit inner loop
                        end
                    end
                end
            end)
        end
        task.wait(0.22) -- General wait to avoid hogging CPU
    end
end)

-- Helper function for pet equipping
local function equipPets(petFolder, action, maxPets)
    local petsToEquip = {}
    for , pet in ipairs(petFolder:GetChildren()) do
        if pet:IsA("Instance") then -- Ensure it's an instance, not a folder
            table.insert(petsToEquip, pet)
        end
    end

    local rEvents = RS:FindFirstChild(GAMECONSTANTS.REVENTSFOLDERNAME)
    local equipPetEvent = rEvents and rEvents:FindFirstChild(GAMECONSTANTS.EQUIPPETEVENTNAME)

    if not equipPetEvent then
        warn("[INV] equipPetEvent not found for pet management.")
        return
    end

    local count = math.min(petsToEquip, maxPets or 8) -- Default to 8 pets if maxPets not specified
    for i = 1, count do
        pcall(function()
            equipPetEvent:FireServer(action, petsToEquip[i])
        end)
        task.wait(0.1) -- Small delay between equipping
    end
end

tabRebirth:AddSwitch("Fast Rebirth Pack Farm", function(value)
    FastPackFarm = value
    if value then
        task.spawn(function()
            local rEvents = RS:FindFirstChild(GAMECONSTANTS.REVENTSFOLDERNAME)
            local equipPetEvent = rEvents and rEvents:FindFirstChild(GAMECONSTANTS.EQUIPPETEVENTNAME)
            local rebirthRemote = rEvents and rEvents:FindFirstChild(GAMECONSTANTS.REBIRTHREMOTENAME)

            if not equipPetEvent or not rebirthRemote then
                warn("[INV] Required remotes not found for Fast Pack Farm. Disabling.")
                FastPackFarm = false
                return
            end

            local petsFolder = LocalPlayer:WaitForChild(GAMECONSTANTS.PETSFOLDERNAME, 10)
            local leaderstats = LocalPlayer:WaitForChild(GAMECONSTANTS.LEADERSTATSNAME, 10)
            local strengthStat = leaderstats:FindFirstChild(GAMECONSTANTS.STRENGTHNAME)
            local rebirthsStat = leaderstats:FindFirstChild(GAMECONSTANTS.REBIRTHSNAME)

            if not petsFolder or not leaderstats or not strengthStat or not rebirthsStat then
                warn("[INV] Required leaderstats or pet folders not found. Disabling Fast Pack Farm.")
                FastPackFarm = false
                return
            end

            while FastPackFarm and NexusRunning do
                pcall(function()
                    -- Equip general pets first
                    equipPets(petsFolder, GAMECONSTANTS.EQUIPPETACTION, 8)
                    task.wait(0.2)

                    -- Calculate required strength for next rebirth
                    local requiredStrength = 10000 + 5000  rebirthsStat.Value
                    local goldenRebirthMultiplier = 1.0
                    local ultimatesFolder = LocalPlayer:FindFirstChild(GAMECONSTANTS.ULTIMATESFOLDERNAME)
                    if ultimatesFolder then
                        local goldenRebirth = ultimatesFolder:FindFirstChild(GAMECONSTANTS.GOLDENREBIRTHNAME)
                        if goldenRebirth and goldenRebirth.Value then
                            goldenRebirthMultiplier = math.max(0, 1 - goldenRebirth.Value  0.1) -- Ensure multiplier doesn't go below 0
                        end
                    end
                    requiredStrength = math.floor(requiredStrength  goldenRebirthMultiplier)

                    -- Equip best pet fallback (if available)
                    local uniquePets = petsFolder:FindFirstChild(GAMECONSTANTS.UNIQUEPETSFOLDERNAME)
                    if uniquePets then
                        local bestPet = uniquePets:FindFirstChild("Swift Samurai") or uniquePets:GetChildren()[1]
                        if bestPet then
                            equipPetEvent:FireServer(GAMECONSTANTS.EQUIPPETACTION, bestPet)
                        end
                    end

                    -- Farm strength until requirement is met
                    while strengthStat.Value < requiredStrength and FastPackFarm and NexusRunning do
                        local muscleEvent = LocalPlayer:WaitForChild(GAMECONSTANTS.MUSCLEEVENTNAME, 5)
                        if muscleEvent then
                            for  = 1, 12 do muscleEvent:FireServer("rep") end
                        else
                            warn("[INV] muscleEvent not found during strength farming.")
                        end
                        task.wait(0.05)
                    end

                    -- Re-equip general pets before rebirth
                    equipPets(petsFolder, GAMECONSTANTS.EQUIPPETACTION, 8)

                    -- Equip strong pet for rebirth (if available)
                    if uniquePets then
                        local strongPet = uniquePets:FindFirstChild("Tribal Overlord") or uniquePets:GetChildren()[1]
                        if strongPet then
                            equipPetEvent:FireServer(GAMECONSTANTS.EQUIPPETACTION, strongPet)
                        end
                    end

                    -- Jungle Bar Lift sit (if applicable)
                    local machinesFolder = WS:FindFirstChild(GAMECONSTANTS.MACHINESFOLDERNAME)
                    if machinesFolder then
                        local jungleBarLift = machinesFolder:FindFirstChild("Jungle Bar Lift")
                        local interactSeat = jungleBarLift and jungleBarLift:FindFirstChild(GAMECONSTANTS.INTERACTSEATNAME)
                        if interactSeat and getHumanoidRootPart() then
                            getHumanoidRootPart().CFrame = interactSeat.CFrame  CFrame.new(0, 4, 0)
                            task.wait(0.4)
                        end
                    end

                    -- Perform rebirth
                    local oldRebirthCount = rebirthsStat.Value
                    repeat
                        rebirthRemote:InvokeServer(GAMECONSTANTS.REBIRTHREQUESTACTION)
                        task.wait(0.15)
                    until rebirthsStat.Value > oldRebirthCount or not FastPackFarm or not NexusRunning
                end)
                task.wait(0.6) -- Wait between rebirth cycles
            end
        end)
    end
end)

-- Optimization and Anti-Lag Features

        for , gui in ipairs(LocalPlayer.PlayerGui:GetChildren()) do
            if gui:IsA(GAMECONSTANTS.SCREENGUITYPE) then gui:Destroy() end
        end
        -- Destroy visual effects in Workspace
        for , obj in ipairs(WS:GetDescendants()) do
            if obj:IsA(GAMECONSTANTS.PARTICLEEMITTERTYPE) or obj:IsA(GAMECONSTANTS.POINTLIGHTTYPE) or
               obj:IsA(GAMECONSTANTS.BLOOMEFFECTTYPE) or obj:IsA(GAMECONSTANTS.BLUREFFECTTYPE) or
               obj:IsA(GAMECONSTANTS.SUNRAYSEFFECTTYPE) then
                obj:Destroy()
            end
        end
        -- Remove Sky object
        for , obj in ipairs(Lighting:GetChildren()) do
            if obj:IsA(GAMECONSTANTS.SKYTYPE) then obj:Destroy() end
        end

        -- Apply dark sky and minimal lighting
        local darkSky = Instance.new(GAMECONSTANTS.SKYTYPE, Lighting)
        for , face in ipairs({"Bk", "Dn", "Ft", "Lf", "Rt", "Up"}) do
            darkSky["Skybox" .. face] = "rbxassetid://0" -- Black skybox
        end
        Lighting.Brightness = 0
        Lighting.ClockTime = 0
        Lighting.OutdoorAmbient = Color3.new()
        Lighting.Ambient = Color3.new()
        Lighting.FogEnd = 120 -- Keep fog for some visibility
        print("[INV] Full optimization applied.")
    end)
end)

-- Position Lock
local positionLockConnection = nil
tabRebirth:AddSwitch("Lock Position", function(enabled)
    PositionLocked = enabled
    local hrp = getHumanoidRootPart()
    if not hrp then return end

    if enabled then
        local currentPosition = hrp.Position
        hrp.Anchored = true -- Anchor the part
        if positionLockConnection then positionLockConnection:Disconnect() end -- Disconnect previous if any

        -- Use Heartbeat to continuously reset position
        positionLockConnection = Run.Heartbeat:Connect(function()
            if PositionLocked and hrp and hrp.Parent then
                hrp.CFrame = CFrame.new(currentPosition)
            end
        end)
        addConnection(positionLockConnection)
        print("[INV] Position locked.")
    else
        if hrp then hrp.Anchored = false end -- Unanchor
        if positionLockConnection then
            positionLockConnection:Disconnect()
            positionLockConnection = nil
        end
        print("[INV] Position unlocked.")
    end
end)

local button = Killer:AddButton("Remove Punch Anim", function()
    local blockedAnimations = {
        ["rbxassetid://3638729053"] = true,
        ["rbxassetid://3638767427"] = true,
    }

    local function setupAnimationBlocking()
        local char = game.Players.LocalPlayer.Character
        if not char or not char:FindFirstChild("Humanoid") then return end

        local humanoid = char:FindFirstChild("Humanoid")

        for _, track in pairs(humanoid:GetPlayingAnimationTracks()) do
            if track.Animation then
                local animId = track.Animation.AnimationId
                local animName = track.Name:lower()

                if blockedAnimations[animId] or
                    animName:match("punch") or
                    animName:match("attack") or
                    animName:match("right") then
                    track:Stop()
                end
            end
        end

        if not _G.AnimBlockConnection then
            local connection = humanoid.AnimationPlayed:Connect(function(track)
                if track.Animation then
                    local animId = track.Animation.AnimationId
                    local animName = track.Name:lower()

                    if blockedAnimations[animId] or
                        animName:match("punch") or
                        animName:match("attack") or
                        animName:match("right") then
                        track:Stop()
                    end
                end
            end)

            _G.AnimBlockConnection = connection
        end
    end

    setupAnimationBlocking()

    local function overrideToolActivation()
        local function processTool(tool)
            if tool and (tool.Name == "Punch" or tool.Name:match("Attack") or tool.Name:match("Right")) then
                if not tool:GetAttribute("ActivatedOverride") then
                    tool:SetAttribute("ActivatedOverride", true)

                    local connection = tool.Activated:Connect(function()
                        task.wait(0.05)

                        local char = game.Players.LocalPlayer.Character
                        if char and char:FindFirstChild("Humanoid") then
                            for _, track in pairs(char.Humanoid:GetPlayingAnimationTracks()) do
                                if track.Animation then
                                    local animId = track.Animation.AnimationId
                                    local animName = track.Name:lower()

                                    if blockedAnimations[animId] or
                                        animName:match("punch") or
                                        animName:match("attack") or
                                        animName:match("right") then
                                        track:Stop()
                                    end
                                end
                            end
                        end
                    end)

                    if not _G.ToolConnections then
                        _G.ToolConnections = {}
                    end
                    _G.ToolConnections[tool] = connection
                end
            end
        end

        for _, tool in pairs(game.Players.LocalPlayer.Backpack:GetChildren()) do
            processTool(tool)
        end

        local char = game.Players.LocalPlayer.Character
        if char then
            for _, tool in pairs(char:GetChildren()) do
                if tool:IsA("Tool") then
                    processTool(tool)
                end
            end
        end

        if not _G.BackpackAddedConnection then
            _G.BackpackAddedConnection = game.Players.LocalPlayer.Backpack.ChildAdded:Connect(function(child)
                if child:IsA("Tool") then
                    task.wait(0.1)
                    processTool(child)
                end
            end)
        end

        if not _G.CharacterToolAddedConnection and char then
            _G.CharacterToolAddedConnection = char.ChildAdded:Connect(function(child)
                if child:IsA("Tool") then
                    task.wait(0.1)
                    processTool(child)
                end
            end)
        end
    end

    overrideToolActivation()

    if not _G.AnimMonitorConnection then
        _G.AnimMonitorConnection = game:GetService("RunService").Heartbeat:Connect(function()
            if tick() % 0.5 < 0.01 then
                local char = game.Players.LocalPlayer.Character
                if char and char:FindFirstChild("Humanoid") then
                    for _, track in pairs(char.Humanoid:GetPlayingAnimationTracks()) do
                        if track.Animation then
                            local animId = track.Animation.AnimationId
                            local animName = track.Name:lower()

                            if blockedAnimations[animId] or
                                animName:match("punch") or
                                animName:match("attack") or
                                animName:match("right") then
                                track:Stop()
                            end
                        end
                    end
                end
            end
        end)
    end

    if not _G.CharacterAddedConnection then
        _G.CharacterAddedConnection = game.Players.LocalPlayer.CharacterAdded:Connect(function(newChar)
            task.wait(1)
            setupAnimationBlocking()
            overrideToolActivation()

            if _G.CharacterToolAddedConnection then
                _G.CharacterToolAdded:Disconnect()
            end

            _G.CharacterToolAdded = newChar.ChildAdded:Connect(function(child)
                if child:IsA("Tool") then
                    task.wait(0.1)
                    processTool(child)
                end
            end)
        end)
    end
end)

function RecoveryPunch()
    if _G.AnimBlockConnection then
        _G.AnimBlockConnection:Disconnect()
        _G.AnimBlockConnection = nil
    end
    if _G.AnimMonitorConnection then
        _G.AnimMonitorConnection:Disconnect()
        _G.AnimMonitorConnection = nil
    end
    if _G.ToolConnections then
        for _, conn in pairs(_G.ToolConnections) do
            if conn then conn:Disconnect() end
        end
        _G.ToolConnections = nil
    end
    if _G.BackpackAddedConnection then
        _G.BackpackAddedConnection:Disconnect()
        _G.BackpackAddedConnection = nil
    end
    if _G.CharacterToolAddedConnection then
        _G.CharacterToolAddedConnection:Disconnect()
        _G.CharacterToolAddedConnection = nil
    end
    if _G.CharacterAddedConnection then
        _G.CharacterAddedConnection:Disconnect()
        _G.CharacterAddedConnection = nil
    end
end
