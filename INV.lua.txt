-- Helper Functions
local function getHumanoidRootPart()
    local character = LocalPlayer.Character
    return character and character:FindFirstChild(GAMECONSTANTS.HUMANOIDROOTPARTNAME)
end

local function getHumanoid()
    local character = LocalPlayer.Character
    return character and character:FindFirstChildOfClass(GAMECONSTANTS.HUMANOIDNAME)
end

local function tpTo(position)
    local hrp = getHumanoidRootPart()
    if hrp and position then
        hrp.CFrame = CFrame.new(position)
    end
end

local function safeEquip(toolName)
    pcall(function()
        local character = LocalPlayer.Character
        local humanoid = getHumanoid()
        if not character or not humanoid or humanoid.Health <= 0 then return end

        local tool = LocalPlayer.Backpack:FindFirstChild(toolName) or character:FindFirstChild(toolName)
        if tool and tool.Parent ~= character then
            character.Humanoid:EquipTool(tool)
        end
    end)
end

-- Auto-Rep (Pushups / Weights / Handstands)
local function startAutoRep(toolName, stateKey)
    task.spawn(function()
        while states[stateKey] and NexusRunning do
            pcall(function()
                local character = LocalPlayer.Character
                local humanoid = getHumanoid()
                if not humanoid or humanoid.Health <= 0 then task.wait(2); return end -- Wait if dead or no humanoid

                safeEquip(toolName)
                local muscleEvent = LocalPlayer:WaitForChild(GAMECONSTANTS.MUSCLEEVENTNAME, 5) -- Wait for event
                if muscleEvent then
                    muscleEvent:FireServer("rep")
                else
                    warn("[INV] muscleEvent not found during auto-rep.")
                    task.wait(1) -- Wait longer if event is missing
                end
            end)
            task.wait(0.1) -- Small delay between actions
        end
        -- Cleanup: Unequip tool if it was equipped by the script
        pcall(function()
            local character = LocalPlayer.Character
            if character then
                local tool = character:FindFirstChild(toolName)
                if tool then tool.Parent = LocalPlayer.Backpack end
            end
        end)
    end)
end

-- Infinite Jump
local jumpConnection = nil
local function toggleInfiniteJump(enabled)
    states.InfiniteJump = enabled
    if enabled then
        if jumpConnection then return end -- Already connected
        jumpConnection = UIS.JumpRequest:Connect(function()
            pcall(function()
                local humanoid = getHumanoid()
                if humanoid then
                    humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
                end
            end)
        end)
        addConnection(jumpConnection)
    else
        if jumpConnection then
            jumpConnection:Disconnect()
            jumpConnection = nil
        end
    end
end

end

task.spawn(function()
    while NexusRunning do
        if rockFarmActive and selectedRockDurability > 0 then
            pcall(function()
                local character = LocalPlayer.Character
                local humanoid = getHumanoid()
                if not humanoid or humanoid.Health <= 0 then task.wait(1.5); return end -- Wait if dead

                local durVal = LocalPlayer:FindFirstChild(GAMECONSTANTS.DURABILITYVALUENAME)
                if not durVal or durVal.Value < selectedRockDurability then
                    task.wait(0.8); return -- Wait if durability is too low
                end

                safeEquip(GAMECONSTANTS.PUNCHTOOLNAME)
                local machinesFolder = WS:FindFirstChild(GAMECONSTANTS.MACHINESFOLDERNAME)
                if machinesFolder then
                    for , obj in ipairs(machinesFolder:GetDescendants()) do
                        if obj.Name == GAMECONSTANTS.NEEDEDDURABILITYNAME and obj:IsA("IntValue") and obj.Value == selectedRockDurability then
                            local rock = obj.Parent:FindFirstChild(GAMECONSTANTS.ROCKNAME)
                            if rock and rock:IsA("BasePart") then
                                local hrp = getHumanoidRootPart()
                                if hrp then
                                    -- Simulate touching the rock to initiate interaction
                                    firetouchinterest(rock, hrp, 0) -- Start touch
                                    task.wait(0.025)
                                    firetouchinterest(rock, hrp, 1) -- End touch
                                    task.wait(0.05) -- Small delay before firing events

                                    local muscleEvent = LocalPlayer:WaitForChild(GAMECONSTANTS.MUSCLEEVENTNAME, 5)
                                    if muscleEvent then
                                        muscleEvent:FireServer("punch", "leftHand")
                                        muscleEvent:FireServer("punch", "rightHand")
                                    else
                                        warn("[INV] muscleEvent not found during rock farm.")
                                    end
                                end
                            end
                            break -- Found the correct rock, exit inner loop
                        end
                    end
                end
            end)
        end
        task.wait(0.22) -- General wait to avoid hogging CPU
    end
end)

-- Helper function for pet equipping
local function equipPets(petFolder, action, maxPets)
    local petsToEquip = {}
    for , pet in ipairs(petFolder:GetChildren()) do
        if pet:IsA("Instance") then -- Ensure it's an instance, not a folder
            table.insert(petsToEquip, pet)
        end
    end

    local rEvents = RS:FindFirstChild(GAMECONSTANTS.REVENTSFOLDERNAME)
    local equipPetEvent = rEvents and rEvents:FindFirstChild(GAMECONSTANTS.EQUIPPETEVENTNAME)

    if not equipPetEvent then
        warn("[INV] equipPetEvent not found for pet management.")
        return
    end

    local count = math.min(petsToEquip, maxPets or 8) -- Default to 8 pets if maxPets not specified
    for i = 1, count do
        pcall(function()
            equipPetEvent:FireServer(action, petsToEquip[i])
        end)
        task.wait(0.1) -- Small delay between equipping
    end
end

tabRebirth:AddSwitch("Fast Rebirth Pack Farm", function(value)
    FastPackFarm = value
    if value then
        task.spawn(function()
            local rEvents = RS:FindFirstChild(GAMECONSTANTS.REVENTSFOLDERNAME)
            local equipPetEvent = rEvents and rEvents:FindFirstChild(GAMECONSTANTS.EQUIPPETEVENTNAME)
            local rebirthRemote = rEvents and rEvents:FindFirstChild(GAMECONSTANTS.REBIRTHREMOTENAME)

            if not equipPetEvent or not rebirthRemote then
                warn("[INV] Required remotes not found for Fast Pack Farm. Disabling.")
                FastPackFarm = false
                return
            end

            local petsFolder = LocalPlayer:WaitForChild(GAMECONSTANTS.PETSFOLDERNAME, 10)
            local leaderstats = LocalPlayer:WaitForChild(GAMECONSTANTS.LEADERSTATSNAME, 10)
            local strengthStat = leaderstats:FindFirstChild(GAMECONSTANTS.STRENGTHNAME)
            local rebirthsStat = leaderstats:FindFirstChild(GAMECONSTANTS.REBIRTHSNAME)

            if not petsFolder or not leaderstats or not strengthStat or not rebirthsStat then
                warn("[INV] Required leaderstats or pet folders not found. Disabling Fast Pack Farm.")
                FastPackFarm = false
                return
            end

            while FastPackFarm and NexusRunning do
                pcall(function()
                    -- Equip general pets first
                    equipPets(petsFolder, GAMECONSTANTS.EQUIPPETACTION, 8)
                    task.wait(0.2)

                    -- Calculate required strength for next rebirth
                    local requiredStrength = 10000 + 5000  rebirthsStat.Value
                    local goldenRebirthMultiplier = 1.0
                    local ultimatesFolder = LocalPlayer:FindFirstChild(GAMECONSTANTS.ULTIMATESFOLDERNAME)
                    if ultimatesFolder then
                        local goldenRebirth = ultimatesFolder:FindFirstChild(GAMECONSTANTS.GOLDENREBIRTHNAME)
                        if goldenRebirth and goldenRebirth.Value then
                            goldenRebirthMultiplier = math.max(0, 1 - goldenRebirth.Value  0.1) -- Ensure multiplier doesn't go below 0
                        end
                    end
                    requiredStrength = math.floor(requiredStrength  goldenRebirthMultiplier)

                    -- Equip best pet fallback (if available)
                    local uniquePets = petsFolder:FindFirstChild(GAMECONSTANTS.UNIQUEPETSFOLDERNAME)
                    if uniquePets then
                        local bestPet = uniquePets:FindFirstChild("Swift Samurai") or uniquePets:GetChildren()[1]
                        if bestPet then
                            equipPetEvent:FireServer(GAMECONSTANTS.EQUIPPETACTION, bestPet)
                        end
                    end

                    -- Farm strength until requirement is met
                    while strengthStat.Value < requiredStrength and FastPackFarm and NexusRunning do
                        local muscleEvent = LocalPlayer:WaitForChild(GAMECONSTANTS.MUSCLEEVENTNAME, 5)
                        if muscleEvent then
                            for  = 1, 12 do muscleEvent:FireServer("rep") end
                        else
                            warn("[INV] muscleEvent not found during strength farming.")
                        end
                        task.wait(0.05)
                    end

                    -- Re-equip general pets before rebirth
                    equipPets(petsFolder, GAMECONSTANTS.EQUIPPETACTION, 8)

                    -- Equip strong pet for rebirth (if available)
                    if uniquePets then
                        local strongPet = uniquePets:FindFirstChild("Tribal Overlord") or uniquePets:GetChildren()[1]
                        if strongPet then
                            equipPetEvent:FireServer(GAMECONSTANTS.EQUIPPETACTION, strongPet)
                        end
                    end

                    -- Jungle Bar Lift sit (if applicable)
                    local machinesFolder = WS:FindFirstChild(GAMECONSTANTS.MACHINESFOLDERNAME)
                    if machinesFolder then
                        local jungleBarLift = machinesFolder:FindFirstChild("Jungle Bar Lift")
                        local interactSeat = jungleBarLift and jungleBarLift:FindFirstChild(GAMECONSTANTS.INTERACTSEATNAME)
                        if interactSeat and getHumanoidRootPart() then
                            getHumanoidRootPart().CFrame = interactSeat.CFrame  CFrame.new(0, 4, 0)
                            task.wait(0.4)
                        end
                    end

                    -- Perform rebirth
                    local oldRebirthCount = rebirthsStat.Value
                    repeat
                        rebirthRemote:InvokeServer(GAMECONSTANTS.REBIRTHREQUESTACTION)
                        task.wait(0.15)
                    until rebirthsStat.Value > oldRebirthCount or not FastPackFarm or not NexusRunning
                end)
                task.wait(0.6) -- Wait between rebirth cycles
            end
        end)
    end
end)

-- Optimization and Anti-Lag Features

        for , gui in ipairs(LocalPlayer.PlayerGui:GetChildren()) do
            if gui:IsA(GAMECONSTANTS.SCREENGUITYPE) then gui:Destroy() end
        end
        -- Destroy visual effects in Workspace
        for , obj in ipairs(WS:GetDescendants()) do
            if obj:IsA(GAMECONSTANTS.PARTICLEEMITTERTYPE) or obj:IsA(GAMECONSTANTS.POINTLIGHTTYPE) or
               obj:IsA(GAMECONSTANTS.BLOOMEFFECTTYPE) or obj:IsA(GAMECONSTANTS.BLUREFFECTTYPE) or
               obj:IsA(GAMECONSTANTS.SUNRAYSEFFECTTYPE) then
                obj:Destroy()
            end
        end
        -- Remove Sky object
        for , obj in ipairs(Lighting:GetChildren()) do
            if obj:IsA(GAMECONSTANTS.SKYTYPE) then obj:Destroy() end
        end

        -- Apply dark sky and minimal lighting
        local darkSky = Instance.new(GAMECONSTANTS.SKYTYPE, Lighting)
        for , face in ipairs({"Bk", "Dn", "Ft", "Lf", "Rt", "Up"}) do
            darkSky["Skybox" .. face] = "rbxassetid://0" -- Black skybox
        end
        Lighting.Brightness = 0
        Lighting.ClockTime = 0
        Lighting.OutdoorAmbient = Color3.new()
        Lighting.Ambient = Color3.new()
        Lighting.FogEnd = 120 -- Keep fog for some visibility
        print("[INV] Full optimization applied.")
    end)
end)

local function autoKarmaLoop(targetKarmaType) -- "goodKarma" or "evilKarma"
    task.spawn(function()
        while (targetKarmaType == "goodKarma" and autoGoodKarma) or (targetKarmaType == "evilKarma" and autoBadKarma) do
            local character = LocalPlayer.Character
            local rightHand = character and character:FindFirstChild(GAMECONSTANTS.RIGHTHANDNAME)
            local leftHand = character and character:FindFirstChild(GAMECONSTANTS.LEFTHANDNAME)

            if character and rightHand and leftHand then
                for , targetPlayer in ipairs(Players:GetPlayers()) do
                    if targetPlayer ~= LocalPlayer then
                        local targetChar = targetPlayer.Character
                        local rootPart = targetChar and targetChar:FindFirstChild(GAMECONSTANTS.HUMANOIDROOTPARTNAME)
                        if rootPart then
                            local evilKarmaVal = targetPlayer:FindFirstChild(GAMECONSTANTS.EVILKARMANAME)
                            local goodKarmaVal = targetPlayer:FindFirstChild(GAMECONSTANTS.GOODKARMANAME)

                            if evilKarmaVal and goodKarmaVal and evilKarmaVal:IsA("IntValue") and goodKarmaVal:IsA("IntValue") then
                                local shouldAttack = false
                                if targetKarmaType == "goodKarma" and goodKarmaVal.Value > evilKarmaVal.Value then
                                    shouldAttack = true -- Target has more good karma, attack to reduce it (or gain evil)
                                elseif targetKarmaType == "evilKarma" and evilKarmaVal.Value > goodKarmaVal.Value then
                                    shouldAttack = true -- Target has more evil karma, attack to reduce it (or gain good)
                                end

                                if shouldAttack then
                                    -- Simulate touch/attack
                                    pcall(function()
                                        firetouchinterest(rightHand, rootPart, 1)
                                        firetouchinterest(leftHand, rootPart, 1)
                                        task.wait(0.02) -- Short delay
                                        firetouchinterest(rightHand, rootPart, 0)
                                        firetouchinterest(leftHand, rootPart, 0)
                                    end)
                                end
                            end
                        end
                    end
                end
            end
            task.wait(0.01) -- Very small wait to keep it responsive
        end
    end)
end

-- Whitelist Friends
local function updateWhitelist(enable)
    if enable then
        -- Add friends to whitelist
        for , player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and LocalPlayer:IsFriendsWith(player.UserId) then
                playerWhitelist[player.Name] = true
            end
        end
        -- Connect to PlayerAdded to catch new friends
        local friendAddedConn = Players.PlayerAdded:Connect(function(player)
            if player ~= LocalPlayer and LocalPlayer:IsFriendsWith(player.UserId) then
                playerWhitelist[player.Name] = true
            end
        end)
        addConnection(friendAddedConn) -- Add to global connections for cleanup
    else
        -- Clear whitelist entries that are friends
        for name in pairs(playerWhitelist) do
            local friend = Players:FindFirstChild(name)
            if friend and LocalPlayer:IsFriendsWith(friend.UserId) then
                playerWhitelist[name] = nil
            end
        end
        -- Note: This doesn't disconnect the PlayerAdded event, which might be a memory leak.
        -- A more robust solution would involve storing and disconnecting specific connections.
        -- For simplicity, assuming the script is reloaded or closed when disabling.
    end
end

killerTab:AddSwitch("Auto Whitelist Friends", function(state)
    updateWhitelist(state)
end)

-- Animation Blocking Logic (No Punch Animation)
local blockedAnimations = {
    [GAMECONSTANTS.DEFAULTPUNCHANIMID] = true,
    [GAMECONSTANTS.DEFAULTATTACKANIMID] = true,
}

local animBlockConnection = nil
local animMonitorConnection = nil
local toolConnections = {}
local backpackAddedConnection = nil
local characterToolAddedConnection = nil
local characterAddedConnection = nil

local function setupAnimationBlocking()
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild(GAMECONSTANTS.HUMANOIDNAME) then return end
    local humanoid = char:FindFirstChild(GAMECONSTANTS.HUMANOIDNAME)

    -- Stop currently playing blocked animations
    for , track in pairs(humanoid:GetPlayingAnimationTracks()) do
        if track.Animation then
            local animId = track.Animation.AnimationId
            local animName = track.Name:lower()
            if blockedAnimations[animId] or animName:match("punch") or animName:match("attack") or animName:match("right") then
                track:Stop()
            end
        end
    end

    -- Connect to AnimationPlayed event
    if not animBlockConnection then
        animBlockConnection = humanoid.AnimationPlayed:Connect(function(track)
            if track.Animation then
                local animId = track.Animation.AnimationId
                local animName = track.Name:lower()
                if blockedAnimations[animId] or animName:match("punch") or animName:match("attack") or animName:match("right") then
                    track:Stop()
                end
            end
        end)
        addConnection(animBlockConnection) -- Add to global connections
    end
end

local function overrideToolActivation()
    local function processTool(tool)
        if tool and (tool.Name == GAMECONSTANTS.PUNCHTOOLNAME or tool.Name:match("Attack") or tool.Name:match("Right")) then
            if not tool:GetAttribute("ActivatedOverride") then
                tool:SetAttribute("ActivatedOverride", true)
                local connection = tool.Activated:Connect(function()
                    task.wait(0.05) -- Small delay
                    local char = LocalPlayer.Character
                    if char and char:FindFirstChild(GAMECONSTANTS.HUMANOIDNAME) then
                        local humanoid = char:FindFirstChild(GAMECONSTANTS.HUMANOIDNAME)
                        for , track in pairs(humanoid:GetPlayingAnimationTracks()) do
                            if track.Animation then
                                local animId = track.Animation.AnimationId
                                local animName = track.Name:lower()
                                if blockedAnimations[animId] or animName:match("punch") or animName:match("attack") or animName:match("right") then
                                    track:Stop()
                                end
                            end
                        end
                    end
                end)
                toolConnections[tool] = connection -- Store connection for this tool
            end
        end
    end

    -- Process tools in backpack and character
    for , tool in pairs(LocalPlayer.Backpack:GetChildren()) do
        if tool:IsA("Tool") then processTool(tool) end
    end
    local char = LocalPlayer.Character
    if char then
        for , tool in pairs(char:GetChildren()) do
            if tool:IsA("Tool") then processTool(tool) end
        end
    end

    -- Connect to ChildAdded for backpack and character
    if not backpackAddedConnection then
        backpackAddedConnection = LocalPlayer.Backpack.ChildAdded:Connect(function(child)
            if child:IsA("Tool") then
                task.wait(0.1) -- Wait for tool to be fully ready
                processTool(child)
            end
        end)
        addConnection(backpackAddedConnection)
    end
    if char and not characterToolAddedConnection then
        characterToolAddedConnection = char.ChildAdded:Connect(function(child)
            if child:IsA("Tool") then
                task.wait(0.1)
                processTool(child)
            end
        end)
        addConnection(characterToolAddedConnection)
    end
end

-- Monitor animations on Heartbeat for robustness

        animMonitorConnection = Run.Heartbeat:Connect(function()

            if tick() % 0.5 < 0.01 then
                local char = LocalPlayer.Character
                if char and char:FindFirstChild(GAMECONSTANTS.HUMANOIDNAME) then
                    local humanoid = char:FindFirstChild(GAMECONSTANTS.HUMANOIDNAME)
                    for , track in pairs(humanoid:GetPlayingAnimationTracks()) do
                        if track.Animation then
                            local animId = track.Animation.AnimationId
                            local animName = track.Name:lower()
                            if blockedAnimations[animId] or animName:match("punch") or animName:match("attack") or animName:match("right") then
                                track:Stop()
                            end
                        end
                    end
                end
            end
        end)
        addConnection(animMonitorConnection)
    end
end

-- Function to recover/remove animation blocking
local function recoverPunchAnimation()
    -- Disconnect all animation-related connections
    if animBlockConnection then animBlockConnection:Disconnect(); animBlockConnection = nil end
    if animMonitorConnection then animMonitorConnection:Disconnect(); animMonitorConnection = nil end
    if characterAddedConnection then characterAddedConnection:Disconnect(); characterAddedConnection = nil end
    if backpackAddedConnection then backpackAddedConnection:Disconnect(); backpackAddedConnection = nil end
    if characterToolAddedConnection then characterToolAddedConnection:Disconnect(); characterToolAddedConnection = nil end

    -- Disconnect tool activation overrides
    for tool, conn in pairs(toolConnections) do
        if conn then conn:Disconnect() end
    end
    toolConnections = {}

    -- Remove global references if they exist (though ideally they shouldn't be used)
    G.AnimBlockConnection = nil
    G.AnimMonitorConnection = nil
    G.ToolConnections = nil
    G.BackpackAddedConnection = nil
    G.CharacterToolAddedConnection = nil
    G.CharacterAddedConnection = nil

    print("[INV] Punch animation blocking removed.")
end

-- Follow System (Teleport Player)
local function startFollowingLoop()
    task.spawn(function()
        while following and NexusRunning do
            if followTarget then
                local targetPlayer = Players:FindFirstChild(followTarget)
                if targetPlayer and targetPlayer ~= LocalPlayer then
                    followPlayer(targetPlayer)
                else
                    -- Target left or is self, stop following
                    following = false
                    followTarget = nil
                    print("[INV] Follow target not found or is self. Stopped following.")
                end
            else
                -- No target set, stop following
                following = false
                print("[INV] No follow target set. Stopped following.")
            end
            task.wait(0.01) -- Update position frequently
        end
    end)
end

local followDropdown = killerTab:AddDropdown("Teleport player", function(selectedDisplayName)
    for , player in ipairs(Players:GetPlayers()) do
        if player.DisplayName == selectedDisplayName then
            followTarget = player.Name
            following = true
            print(string.format("[INV] Started following: %s", player.Name))
            -- Teleport immediately
            local targetPlayer = Players:FindFirstChild(followTarget)
            if targetPlayer then followPlayer(targetPlayer) end
            startFollowingLoop() -- Ensure loop is running
            break
        end
    end
end)

-- Populate follow dropdown
local function populateFollowDropdown()
    followDropdown:Clear()
    for , player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            followDropdown:Add(player.DisplayName)
        end
    end
end
populateFollowDropdown()

-- Auto Slam
local function autoSlam()
    local groundSlamTool = LocalPlayer.Backpack:FindFirstChild(GAMECONSTANTS.GROUNDSLAMTOOLNAME) or (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild(GAMECONSTANTS.GROUNDSLAMTOOLNAME))
    if groundSlamTool then
        if groundSlamTool.Parent == LocalPlayer.Backpack then
            groundSlamTool.Parent = LocalPlayer.Character
        end
        -- Reset attack time if it exists
        local attackTime = groundSlamTool:FindFirstChild("attackTime")
        if attackTime and attackTime.Value then
            attackTime.Value = 0
        end
        local muscleEvent = LocalPlayer:WaitForChild(GAMECONSTANTS.MUSCLEEVENTNAME, 5)
        if muscleEvent then
            muscleEvent:FireServer("slam")
        else
            warn("[INV] muscleEvent not found for auto slam.")
        end
        groundSlamTool:Activate()
    else
        warn("[INV] Ground Slam tool not found.")
        return false -- Indicate failure
    end
    return true -- Indicate success
end

killerTab:AddSwitch("auto slams", function(value)
    if value then
        task.spawn(function()
            while value and NexusRunning do
                if not autoSlam() then
                    -- If autoSlam failed (e.g., tool not found), disable the switch
                    value = false
                    killerTab:GetElement("auto slams"):SetValue(false) -- Update UI
                    print("[INV] Auto Slam disabled due to tool not found.")
                end
                task.wait(0.1)
            end
        end)
    end
end)
