-- AUTO RELOAD TOOL FUNCTION
local function startAutoRep(toolName, toggleVar)
    task.spawn(function()
        while _G[toggleVar] do
            pcall(function()
                local char = LocalPlayer.Character
                if not char or not char:FindFirstChild("Humanoid") or char.Humanoid.Health <= 0 then
                    task.wait(1.5)
                    return
                end
                local tool = LocalPlayer.Backpack:FindFirstChild(toolName) or char:FindFirstChild(toolName)
                if tool and tool.Parent ~= char then
                    char.Humanoid:EquipTool(tool)
                end
                if muscleEvent then
                    muscleEvent:FireServer("rep")
                end
            end)
            task.wait(0.025)
        end
        -- Unequip tool when toggled off
        pcall(function()
            local char = LocalPlayer.Character
            if char then
                local eq = char:FindFirstChild(toolName)
                if eq then eq.Parent = LocalPlayer.Backpack end
            end
        end)
    end)
end

-- KILL TAB
local KillTab = window:AddTab("Kill")
local KillFolder = KillTab:AddFolder("Auto Kill")

local function checkCharacter()
    if not LocalPlayer.Character then
        LocalPlayer.CharacterAdded:Wait()
    end
    return LocalPlayer.Character
end

local function killPlayer(target)
    local char = checkCharacter()
    if not char or not char:FindFirstChild("LeftHand") then return end
    local tChar = target.Character
    if tChar and tChar:FindFirstChild("HumanoidRootPart") and tChar:FindFirstChild("Humanoid") and tChar.Humanoid.Health > 0 then
        firetouchinterest(tChar.HumanoidRootPart, char.LeftHand, 0)
        firetouchinterest(tChar.HumanoidRootPart, char.LeftHand, 1)
        local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or char:FindFirstChild("Punch")
        if punch and char:FindFirstChild("Humanoid") then
            char.Humanoid:EquipTool(punch)
        end
        if muscleEvent then
            muscleEvent:FireServer("punch", "leftHand")
            muscleEvent:FireServer("punch", "rightHand")
        end
    end
end

local function safeEquipTool(toolName)
    pcall(function()
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("Humanoid") then
            for _, tool in pairs(LocalPlayer.Backpack:GetChildren()) do
                if tool.Name == toolName then
                    char.Humanoid:EquipTool(tool)
                end
            end
            if muscleEvent then
                muscleEvent:FireServer("punch", "leftHand")
                muscleEvent:FireServer("punch", "rightHand")
            end
        end
    end)
end

local rockFarmActive = false
local function setRockFarm(durability, enable)
    if enable then
        getgenv().selectedRockDurability = durability
        getgenv().rockFarmActive = true
    else
        if getgenv().selectedRockDurability == durability then
            getgenv().rockFarmActive = false
            getgenv().selectedRockDurability = 0
        end
    end
end

-- Rock Farming Loop
task.spawn(function()
    while true do
        if not getgenv().rockFarmActive or not getgenv().selectedRockDurability or getgenv().selectedRockDurability <= 0 then
            task.wait(0.8)
        else
            pcall(function()
                local char = LocalPlayer.Character
                if not char or not char:FindFirstChild("Humanoid") or char.Humanoid.Health <= 0 then
                    task.wait(1.4)
                    return
                end

                local dur = LocalPlayer:FindFirstChild("Durability")
                if not dur or dur.Value < getgenv().selectedRockDurability then
                    task.wait(0.6)
                    return
                end

                safeEquipTool("Punch")
                local machinesFolder = workspace:FindFirstChild("machinesFolder")
                if machinesFolder then
                    for _, obj in pairs(machinesFolder:GetDescendants()) do
                        if obj.Name == "neededDurability" and obj:IsA("IntValue") and obj.Value == getgenv().selectedRockDurability then
                            local rock = obj.Parent:FindFirstChild("Rock")
                            if rock and rock:IsA("BasePart") then
                                local hrp = char:FindFirstChild("HumanoidRootPart")
                                if hrp then
                                    if char:FindFirstChild("RightHand") then
                                        firetouchinterest(rock, hrp, 0)
                                        task.wait(0.03)
                                        firetouchinterest(rock, hrp, 1)
                                    end
                                    if char:FindFirstChild("LeftHand") then
                                        firetouchinterest(rock, hrp, 0)
                                        task.wait(0.03)
                                        firetouchinterest(rock, hrp, 1)
                                    end
                                    -- Punch
                                    if muscleEvent then
                                        muscleEvent:FireServer("punch", "leftHand")
                                        muscleEvent:FireServer("punch", "rightHand")
                                    end
                                end
                            end
                        end
                    end
                end
            end)
            task.wait(0.18)
        end
    end
end)

local rebirthTab = window:AddTab("rebirth")
rebirthTab:AddSwitch("Auto Rebirth (Infinite)", function(state)
    getgenv().autoRebirth = state
    if state then
        getgenv().autoRebirthLoop = task.spawn(function()
            while getgenv().autoRebirth do
                pcall(function()
                    local rEvents = ReplicatedStorage:WaitForChild("rEvents", 5)
                    if rEvents then
                        local rebirthRemote = rEvents:FindFirstChild("rebirthRemote")
                        if rebirthRemote then
                            rebirthRemote:InvokeServer("rebirthRequest")
                        else
                            warn("rebirthRemote not found - check ReplicatedStorage.rEvents")
                        end
                    end
                end)
                task.wait(0.1)
            end
        end)
        print("Auto Rebirth ENABLED")
    else
        getgenv().autoRebirth = false
    end
end)
