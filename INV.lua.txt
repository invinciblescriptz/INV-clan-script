-- Fast Punch switch
Tools:AddSwitch("Fast Punch", function(state)
    _G.FastPunch = state
    if state then
        task.spawn(function()
            while _G.FastPunch do
                pcall(function()
                    -- Get the muscleEvent object
                    local muscleEvent = game.ReplicatedStorage:FindFirstChild("muscleEvent", true)
                    if muscleEvent then
                        muscleEvent:FireServer("punch", "rightHand")
                        muscleEvent:FireServer("punch", "leftHand")
                    end

                    local player = game.Players.LocalPlayer
                    if player and player.Character and player.Character:FindFirstChild("Punch") and player.Character.Punch:IsA("Tool") then
                        player.Character.Punch:Activate()
                    end
                end)
                task.wait(0.07)
            end
        end)
    end
end)

           v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("DepthOfFieldEffect") or v:IsA("SunRaysEffect") then
            v.Enabled = false
        end
    end

    settings().Rendering.QualityLevel = Enum.QualityLevel.Level01

-- Free AutoLift Gamepass button
Tools:AddButton("Free AutoLift Gamepass", function()
    pcall(function()
        local gpFolder = game.ReplicatedStorage:FindFirstChild("gamepassIds")
        if gpFolder then
            local player = game.Players.LocalPlayer
            local owned = player:FindFirstChild("ownedGamepasses")
            if not owned then
                owned = Instance.new("Folder")
                owned.Name = "ownedGamepasses"
                owned.Parent = player
            end

            for _, gp in pairs(gpFolder:GetChildren()) do
                if gp:IsA("IntValue") or gp:IsA("NumberValue") then
                    local val = Instance.new("IntValue")
                    val.Name = gp.Name
                    val.Value = gp.Value
                    val.Parent = owned
                end
            end
            print("Gamepasses added (may require relog / rejoin)")
        else
            print("gamepassIds folder not found in ReplicatedStorage")
        end
    end)
end)
