-- Helper functions
local function getHRP()
    local char = LP.Character
    return char and char:FindFirstChild("HumanoidRootPart")
end

local function tpTo(pos)
    local hrp = getHRP()
    if hrp then hrp.CFrame = CFrame.new(pos) end
end

local function safeEquip(toolName)
    pcall(function()
        local char = LP.Character
        if not char or not char:FindFirstChild("Humanoid") then return end
        local t = LP.Backpack:FindFirstChild(toolName) or char:FindFirstChild(toolName)
        if t and t.Parent ~= char then
            char.Humanoid:EquipTool(t)
        end
    end)
end

-- Auto-rep function
local function startAutoRep(toolName, key)
    task.spawn(function()
        while states[key] and getgenv().NexusRunning do
            pcall(function()
                local char = LP.Character
                local h = char and char:FindFirstChildOfClass("Humanoid")
                if not h or h.Health <= 0 then task.wait(2) return end
                safeEquip(toolName)
                muscleEvent:FireServer("rep")
            end)
            task.wait(0.1)
        end
        -- cleanup
        pcall(function()
            local t = LP.Character and LP.Character:FindFirstChild(toolName)
            if t then t.Parent = LP.Backpack end
        end)
    end)
end

-- Connection management
local conns = {}
local function addConn(c)
    table.insert(conns, c)
end

-- Infinite Jump
local jumpConn
local function toggleInfJump(en)
    states.InfiniteJump = en
    if en then
        if jumpConn then return end
        jumpConn = UIS.JumpRequest:Connect(function()
            pcall(function()
                local h = LP.Character and LP.Character:FindFirstChildOfClass("Humanoid")
                if h then h:ChangeState(Enum.HumanoidStateType.Jumping) end
            end)
        end)
        addConn(jumpConn)
    else
        if jumpConn then jumpConn:Disconnect() jumpConn = nil end
    end
end

-- Anti-AFK

        if afkConn then afkConn:Disconnect() afkConn = nil end
    end
end

local killAura = false
local killAuraToggle = false
local killAuraConnection = nil

tabKiller:AddSwitch("Kill All Players (Aura)", function(state)
    killAura = state
    if state then
        killAuraToggle = true
        killAuraConnection = task.spawn(function()
            while killAuraToggle and getgenv().NexusRunning do
                pcall(function()
                    local myHRP = getHRP()
                    if not myHRP then return end
                    for _, plr in ipairs(Players:GetPlayers()) do
                        if plr ~= LP and plr.Character then
                            local theirHRP = plr.Character:FindFirstChild("HumanoidRootPart")
                            local theirHum = plr.Character:FindFirstChildOfClass("Humanoid")
                            if theirHRP and theirHum and theirHum.Health > 0 then
                                local dist = (theirHRP.Position - myHRP.Position).Magnitude
                                if dist < 18 then
                                    firetouchinterest(myHRP, theirHRP, 0)
                                    task.wait(0.025)
                                    firetouchinterest(myHRP, theirHRP, 1)
                                    -- Auto spam punch if enabled
                                    if states.FastPunch then
                                        muscleEvent:FireServer("punch", "leftHand")
                                        muscleEvent:FireServer("punch", "rightHand")
                                    end
                                end
                            end
                        end
                    end
                end)
                task.wait(0.08)
            end
            killAuraConnection = nil
        end)
    else
        killAuraToggle = false
        -- No explicit disconnect needed if using flag
    end
end)

-- Target Player Lock & Auto-Kill
local selectedTarget = nil
local targetKillRunning = false
local targetKillTask = nil

tabKiller:AddTextbox("Target Player Name", "", function(txt)
    selectedTarget = nil
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr.Name:lower() == txt:lower() or (plr.DisplayName and plr.DisplayName:lower() == txt:lower()) then
            selectedTarget = plr
            VU:Notify("Target locked → " .. plr.Name, 3, Color3.fromRGB(0, 255, 0))
            return
        end
    end
    VU:Notify("Player not found!", 3, Color3.fromRGB(255, 0, 0))
end)

tabKiller:AddButton("🔒 Lock & Auto-Kill Selected Target", function()
    if not selectedTarget then
        VU:Notify("No target selected!", 3, Color3.fromRGB(255, 0, 0))
        return
    end
    
    targetKillRunning = not targetKillRunning
    if targetKillRunning then
        targetKillTask = task.spawn(function()
            while targetKillRunning and selectedTarget and getgenv().NexusRunning do
                pcall(function()
                    if selectedTarget.Character then
                        local targetHRP = selectedTarget.Character:FindFirstChild("HumanoidRootPart")
                        local myHRP = getHRP()
                        if targetHRP and myHRP then
                            local dist = (targetHRP.Position - myHRP.Position).Magnitude
                            if dist < 60 then
                                myHRP.CFrame = targetHRP.CFrame * CFrame.new(0, 0, -3)
                                firetouchinterest(myHRP, targetHRP, 0)
                                task.wait(0.02)
                                firetouchinterest(myHRP, targetHRP, 1)
                                muscleEvent:FireServer("punch", "leftHand")
                                muscleEvent:FireServer("punch", "rightHand")
                                muscleEvent:FireServer("slam")
                            end
                        end
                    end
                end)
                task.wait(0.06)
            end
        end)
        VU:Notify("AUTO KILLING " .. selectedTarget.Name .. " (toggle again to stop)", 4, Color3.fromRGB(255, 100, 0))
    else
        targetKillRunning = false
        VU:Notify("Target kill stopped", 3)
    end
end)
